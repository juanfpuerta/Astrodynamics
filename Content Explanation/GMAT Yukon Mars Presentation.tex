\begin{frame}{GMAT Strategy: Forward/Backward Integration}
    \begin{columns}
        \begin{column}{0.5\textwidth}
            \textbf{Forward Propagator:}
            \begin{itemize}
                \item Starts at Earth.
                \item Propagates toward the transfer midpoint.
            \end{itemize}
        \end{column}
        \begin{column}{0.5\textwidth}
            \textbf{Backward Propagator:}
            \begin{itemize}
                \item Starts at Mars.
                \item Propagates \textit{backwards} in time.
            \end{itemize}
        \end{column}
    \end{columns}
    \vspace{0.4cm}
    \begin{alertblock}{The Optimization Goal}
        Yukon Optimizer varies the initial states until the two paths have zero position/velocity mismatch at the midpoint.
    \end{alertblock}
\end{frame}

\begin{frame}{The Optimizer's Task}
    \begin{itemize}
        \item \textbf{Vary:} Initial Earth-Departure and Mars-Arrival state vectors.
        \item \textbf{NonLinearConstraint:} Bridge the gap (Match position/velocity).
        \item \textbf{Minimize:} The $\Delta V$ magnitude (or the norm of the velocity discontinuity).
    \end{itemize}
\end{frame}
